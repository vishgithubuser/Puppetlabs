## Using condiftional statements in puppet code

# using Modern facts

if $facts['os']['family'] == 'Debian' {
  package { 'nginx':
    ensure => 'installed',
    provider => 'apt',
  }
}  

## using if-elseif-else

if $facts['os']['family'] == 'RedHat' {
  notify { 'RedHat' : }
}
elsif $facts['os']['family'] == 'Debian' {
  notify { 'Debian' : }
}
else {
  fail ("your OS ${facts['os']['family']} is not supported")
}
